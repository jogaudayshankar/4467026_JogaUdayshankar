#include<stdio.h>

void minimaxsum(int arr[], int n){
    long long total =0;
    int min=arr[0], max=arr[0];
    
    for(int i=0;i<n;i++){
        total+=arr[i];
        if(arr[i]< min)
           min=arr[i];
        if(arr[i]> max)
           max=arr[i];
    }
    long long minsum=total-max;
    long long maxsum=total-min;
    
    printf("%lld %lld", minsum,maxsum);
    
}

int main(){
    int arr[5];
    for(int i=0;i<5;i++){
        scanf("%d",&arr[i]);
    }
    minimaxsum(arr,5);
    return 0;
}
